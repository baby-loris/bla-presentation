Декларативность

            module.exports = new ApiMethod('hello')
                .setDescription('Returns greeting from server')
                .addParam({
                    name: 'name',
                    description: 'User name',
                    required: true
                })
                .setAction(function (params) {
                    return 'Hello, ' + params.name;
                });
